Fonction verificationSaisie( <chaîne> de type chaîne de caractères ) 
	Définir une variable <lngChaîne> de type entier
	Définir une variable <i> de type entier initialisée à 0
	Définir une variable <c> de type caractère
	
	lngChaîne <- longueur(chaîne) // Nombre de caractères de la chaîne passée en paramètre
	
	Pour i variant 0 de lngChaîne-1
		c <- ième caractère(chaîne)
		Si (c < '0' ou c > '9') et (c différent '.') et (c différent ',')
			Renvoyer -1
		Fin de Si
	Fin Pour
	Renvoyer conversion(chaîne)
Fin Fonction	

// Deuxième version (avec un Tant que)
Fonction verificationSaisie( <chaîne> de type chaîne de caractères ) 
	Définir une variable <lngChaîne> de type entier
	Définir une variable <i> de type entier 
	Définir une variable <c> de type caractère
	
	lngChaîne <- longueur(chaîne) // Nombre de caractères de la chaîne passée en paramètre
	i <- 0
	Tant que i < lngChaîne
		c <- ième caractère(chaîne)
		Si (c < '0' ou c > '9') et (c différent '.') et (c différent ',')
			Renvoyer -1
		Sinon
			i <- i+1
		Fin de Si
	Fin de Tant que
	Renvoyer conversion(chaîne)
Fin Fonction
